public class WeaponHandsToken extends AbstractStringStoringToken<CodeControl>
{

	public String getTokenName()

		return CControl.WEAPONHANDS;
	}

	@Override
	public Class<CodeControl> getTokenClass()
	{
		return CodeControl.class;
	}
}
